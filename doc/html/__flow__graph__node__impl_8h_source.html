205  void handle_operations(my_operation *op_list) {
206  my_operation *tmp;
207  while (op_list) {
208  tmp = op_list;
209  op_list = op_list->next;
210  switch (tmp->type) {
211  case reg_pred:
212  my_predecessors.add(*(tmp->r));
213  __TBB_store_with_release(tmp->status, SUCCEEDED);
214  if (!forwarder_busy) {
215  forwarder_busy = true;
216  spawn_forward_task();
217  }
218  break;
219  case rem_pred:
220  my_predecessors.remove(*(tmp->r));
221  __TBB_store_with_release(tmp->status, SUCCEEDED);
222  break;
223  case app_body:
224  __TBB_ASSERT(my_max_concurrency != 0, NULL);
225  --my_concurrency;
226  __TBB_store_with_release(tmp->status, SUCCEEDED);
227  if (my_concurrency<my_max_concurrency) {
228  input_type i;
229  bool item_was_retrieved = false;
230  if ( my_queue )
231  item_was_retrieved = my_queue->pop(i);
232  else
233  item_was_retrieved = my_predecessors.get_item(i);
234  if (item_was_retrieved) {
235  ++my_concurrency;
236  spawn_body_task(i);
237  }
238  }
239  break;
240  case app_body_bypass: {
241  task * new_task = NULL;
242  __TBB_ASSERT(my_max_concurrency != 0, NULL);
243  --my_concurrency;
244  if (my_concurrency<my_max_concurrency) {
245  input_type i;
246  bool item_was_retrieved = false;
247  if ( my_queue )
248  item_was_retrieved = my_queue->pop(i);
249  else
250  item_was_retrieved = my_predecessors.get_item(i);
251  if (item_was_retrieved) {
252  ++my_concurrency;
253  new_task = create_body_task(i);
254  }
255  }
256  tmp->bypass_t = new_task;
257  __TBB_store_with_release(tmp->status, SUCCEEDED);
258  }
259  break;
260  case tryput_bypass: internal_try_put_task(tmp); break;
261  case try_fwd: internal_forward(tmp); break;
262 #if TBB_PREVIEW_FLOW_GRAPH_FEATURES
263  case add_blt_pred: {
264  my_predecessors.internal_add_built_predecessor(*(tmp->r));
265  __TBB_store_with_release(tmp->status, SUCCEEDED);
266  }
267  break;
268  case del_blt_pred:
269  my_predecessors.internal_delete_built_predecessor(*(tmp->r));
270  __TBB_store_with_release(tmp->status, SUCCEEDED);
271  break;
272  case blt_pred_cnt:
273  tmp->cnt_val = my_predecessors.predecessor_count();
274  __TBB_store_with_release(tmp->status, SUCCEEDED);
275  break;
276  case blt_pred_cpy:
277  my_predecessors.copy_predecessors( *(tmp->predv) );
278  __TBB_store_with_release(tmp->status, SUCCEEDED);
279  break;
280 #endif /* TBB_PREVIEW_FLOW_GRAPH_FEATURES */
281  }
282  }
283  }

286  void internal_try_put_task(my_operation *op) {
287  __TBB_ASSERT(my_max_concurrency != 0, NULL);
288  if (my_concurrency < my_max_concurrency) {
289  ++my_concurrency;
290  task * new_task = create_body_task(*(op->elem));
291  op->bypass_t = new_task;
292  __TBB_store_with_release(op->status, SUCCEEDED);
293  } else if ( my_queue && my_queue->push(*(op->elem)) ) {
294  op->bypass_t = SUCCESSFULLY_ENQUEUED;
295  __TBB_store_with_release(op->status, SUCCEEDED);
296  } else {
297  op->bypass_t = NULL;
298  __TBB_store_with_release(op->status, FAILED);
299  }
300  }

303  void internal_forward(my_operation *op) {
304  op->bypass_t = NULL;
305  if (my_concurrency<my_max_concurrency || !my_max_concurrency) {
306  input_type i;
307  bool item_was_retrieved = false;
308  if ( my_queue )
309  item_was_retrieved = my_queue->pop(i);
310  else
311  item_was_retrieved = my_predecessors.get_item(i);
312  if (item_was_retrieved) {
313  ++my_concurrency;
314  op->bypass_t = create_body_task(i);
315  __TBB_store_with_release(op->status, SUCCEEDED);
316  return;
317  }
318  }
319  __TBB_store_with_release(op->status, FAILED);
320  forwarder_busy = false;
321  }

324  // then decides if more work is available
325  void apply_body( input_type &i ) {
326  task *new_task = apply_body_bypass(i);
327  if(!new_task) return;
328  if(new_task == SUCCESSFULLY_ENQUEUED) return;
329  FLOW_SPAWN(*new_task);
330  return;
331  }
332 
334  // then decides if more work is available
335  task * apply_body_bypass( input_type &i ) {
336  task * new_task = static_cast<ImplType *>(this)->apply_body_impl_bypass(i);
337  if ( my_max_concurrency != 0 ) {
338  my_operation op_data(app_body_bypass); // tries to pop an item or get_item, enqueues another apply_body
339  my_aggregator.execute(&op_data);
340  tbb::task *ttask = op_data.bypass_t;
341  new_task = combine_tasks(new_task, ttask);
342  }
343  return new_task;
344  }

357  inline void spawn_body_task( const input_type &input ) {
358  task* tp = create_body_task(input);

366  task *forward_task() {
367  my_operation op_data(try_fwd);
368  task *rval = NULL;
369  do {
370  op_data.status = WAIT;
371  my_aggregator.execute(&op_data);
372  if(op_data.status == SUCCEEDED) {
373  tbb::task *ttask = op_data.bypass_t;
374  rval = combine_tasks(rval, ttask);
375  }
376  } while (op_data.status == SUCCEEDED);
377  return rval;
378  }
