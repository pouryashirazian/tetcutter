231  // Inserts the new element in the list in an atomic fashion
232  nodeptr_t atomic_set_next(nodeptr_t new_node, nodeptr_t current_node)
233  {
234  // Try to change the next pointer on the current element to a new element, only if it still points to the cached next
235  nodeptr_t exchange_node = tbb::internal::as_atomic(my_next).compare_and_swap(new_node, current_node);
236 
237  if (exchange_node == current_node) // TODO: why this branch?
238  {
239  // Operation succeeded, return the new node
240  return new_node;
241  }
242  else
243  {
244  // Operation failed, return the "interfering" node

246  }

490  // Insert a new element between passed in iterators
491  std::pair<iterator, bool> try_insert(raw_iterator it, raw_iterator next, nodeptr_t pnode, size_type *new_count)
492  {
493  nodeptr_t inserted_node = try_insert_atomic(it.get_node_ptr(), pnode, next.get_node_ptr());

508  // Insert a new dummy element, starting search at a parent dummy element
509  raw_iterator insert_dummy(raw_iterator it, sokey_t order_key)
510  {
511  raw_iterator last = raw_end();
512  raw_iterator where = it;
513 
514  __TBB_ASSERT(where != last, "Invalid head node");
515 
516  ++where;
517 
518  // Create a dummy element up front, even though it may be discarded (due to concurrent insertion)
519  nodeptr_t dummy_node = create_node(order_key);
520 
521  for (;;)
522  {
523  __TBB_ASSERT(it != last, "Invalid head list node");
524 
525  // If the head iterator is at the end of the list, or past the point where this dummy
526  // node needs to be inserted, then try to insert it.
527  if (where == last || get_order_key(where) > order_key)
528  {
529  __TBB_ASSERT(get_order_key(it) < order_key, "Invalid node order in the list");
530 
531  // Try to insert it in the right place
532  nodeptr_t inserted_node = try_insert_atomic(it.get_node_ptr(), dummy_node, where.get_node_ptr());
533 
534  if (inserted_node == dummy_node)
535  {
536  // Insertion succeeded, check the list for order violations
537  check_range(it, where);
538  return raw_iterator(dummy_node);
539  }
540  else
541  {
542  // Insertion failed: either dummy node was inserted by another thread, or
543  // a real element was inserted at exactly the same place as dummy node.
544  // Proceed with the search from the previous location where order key was
545  // known to be larger (note: this is legal only because there is no safe
546  // concurrent erase operation supported).
547  where = it;
548  ++where;
549  continue;
550  }
551  }
552  else if (get_order_key(where) == order_key)
553  {
554  // Another dummy node with the same value found, discard the new one.
555  destroy_node(dummy_node);
556  return where;
557  }
558 
559  // Move the iterator forward
560  it = where;
561  ++where;
562  }
563 
564  }

587  // Move all elements from the passed in split-ordered list to this one
588  void move_all(self_type& source)
589  {
590  raw_const_iterator first = source.raw_begin();
591  raw_const_iterator last = source.raw_end();
592 
593  if (first == last)
594  return;
595 
596  nodeptr_t previous_node = my_head;
597  raw_const_iterator begin_iterator = first++;
598 
599  // Move all elements one by one, including dummy ones
600  for (raw_const_iterator it = first; it != last;)
601  {
602  nodeptr_t pnode = it.get_node_ptr();
603 
604  nodeptr_t dummy_node = pnode->is_dummy() ? create_node(pnode->get_order_key()) : create_node(pnode->get_order_key(), pnode->my_element);
605  previous_node = try_insert_atomic(previous_node, dummy_node, NULL);
606  __TBB_ASSERT(previous_node != NULL, "Insertion must succeed");
607  raw_const_iterator where = it++;
608  source.erase_node(get_iterator(begin_iterator), where);
609  }
610  check_range();
611  }
