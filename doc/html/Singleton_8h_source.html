147  template <typename T, typename Destroyer>
148  void SetLongevity(T* pDynObject, unsigned int longevity,
149  Destroyer d)
150  {
151  using namespace Private;
152 
153  // manage lifetime of stack manually
154  if(pTrackerArray==0)
155  pTrackerArray = new TrackerArray;
156 
157  // automatically delete the ConcreteLifetimeTracker object when a exception is thrown
158  std::auto_ptr<LifetimeTracker>
159  p( new ConcreteLifetimeTracker<T, Destroyer>(pDynObject, longevity, d) );
160 
161  // Find correct position
162  TrackerArray::iterator pos = std::upper_bound(
163  pTrackerArray->begin(),
164  pTrackerArray->end(),
165  p.get(),
166  LifetimeTracker::Compare);
167 
168  // Insert the pointer to the ConcreteLifetimeTracker object into the queue
169  pTrackerArray->insert(pos, p.get());
170 
171  // nothing has thrown: don't delete the ConcreteLifetimeTracker object
172  p.release();
173 
174  // Register a call to AtExitFn
175  std::atexit(Private::AtExitFn);
176  }

180  template <typename T, typename Destroyer>
181  void SetLongevity(T* pDynObject, unsigned int longevity,
182  Destroyer d)
183  {
184  using namespace Private;
185 
186  TrackerArray pNewArray = static_cast<TrackerArray>(
187  std::realloc(pTrackerArray,
188  sizeof(*pTrackerArray) * (elements + 1)));
189  if (!pNewArray) throw std::bad_alloc();
190 
191  // Delayed assignment for exception safety
192  pTrackerArray = pNewArray;
193 
194  LifetimeTracker* p = new ConcreteLifetimeTracker<T, Destroyer>(
195  pDynObject, longevity, d);
196 
197  // Insert a pointer to the object into the queue
198  TrackerArray pos = std::upper_bound(
199  pTrackerArray,
200  pTrackerArray + elements,
201  p,
202  LifetimeTracker::Compare);
203  std::copy_backward(
204  pos,
205  pTrackerArray + elements,
206  pTrackerArray + elements + 1);
207  *pos = p;
208  ++elements;
209 
210  // Register a call to AtExitFn
211  std::atexit(Private::AtExitFn);
212  }
