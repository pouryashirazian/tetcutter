86 template< typename T, typename... Types >
87 struct stored_pack<T, Types...> : stored_pack<Types...>
88 {
89  typedef stored_pack<T, Types...> pack_type;
90  typedef stored_pack<Types...> pack_remainder;
91  // Since lifetime of original values is out of control, copies should be made.
92  // Thus references should be stripped away from the deduced type.
93  typename strip<T>::type leftmost_value;
94 
95  // Here rvalue references act in the same way as forwarding references,
96  // as long as class template parameters were deduced via forwarding references.
97  stored_pack( T&& t, Types&&... types )
98  : pack_remainder(std::forward<Types>(types)...), leftmost_value(std::forward<T>(t)) {}
99 
100  // Friend front-end functions
101  template< typename F, typename Pack > friend void call( F&& f, Pack&& p );
102  template< typename Ret, typename F, typename Pack > friend Ret call_and_return( F&& f, Pack&& p );
103 
104 protected:
105  template< typename Ret, typename F, typename... Preceding >
106  static Ret call( F&& f, const pack_type& pack, Preceding&&... params ) {
107  return pack_remainder::template call<Ret>(
108  std::forward<F>(f), static_cast<const pack_remainder&>(pack),
109  std::forward<Preceding>(params)... , pack.leftmost_value
110  );
111  }
112  template< typename Ret, typename F, typename... Preceding >
113  static Ret call( F&& f, pack_type&& pack, Preceding&&... params ) {
114  return pack_remainder::template call<Ret>(
115  std::forward<F>(f), static_cast<pack_remainder&&>(pack),
116  std::forward<Preceding>(params)... , std::move(pack.leftmost_value)
117  );
118  }
119 };
