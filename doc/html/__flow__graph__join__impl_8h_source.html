238  void handle_operations(reserving_port_operation* op_list) {
239  reserving_port_operation *current;
240  bool no_predecessors;
241  while(op_list) {
242  current = op_list;
243  op_list = op_list->next;
244  switch(current->type) {
245  case reg_pred:
246  no_predecessors = my_predecessors.empty();
247  my_predecessors.add(*(current->my_pred));
248  if ( no_predecessors ) {
249  (void) my_join->decrement_port_count(true); // may try to forward
250  }
251  __TBB_store_with_release(current->status, SUCCEEDED);
252  break;
253  case rem_pred:
254  my_predecessors.remove(*(current->my_pred));
255  if(my_predecessors.empty()) my_join->increment_port_count();
256  __TBB_store_with_release(current->status, SUCCEEDED);
257  break;
258  case res_item:
259  if ( reserved ) {
260  __TBB_store_with_release(current->status, FAILED);
261  }
262  else if ( my_predecessors.try_reserve( *(current->my_arg) ) ) {
263  reserved = true;
264  __TBB_store_with_release(current->status, SUCCEEDED);
265  } else {
266  if ( my_predecessors.empty() ) {
267  my_join->increment_port_count();
268  }
269  __TBB_store_with_release(current->status, FAILED);
270  }
271  break;
272  case rel_res:
273  reserved = false;
274  my_predecessors.try_release( );
275  __TBB_store_with_release(current->status, SUCCEEDED);
276  break;
277  case con_res:
278  reserved = false;
279  my_predecessors.try_consume( );
280  __TBB_store_with_release(current->status, SUCCEEDED);
281  break;
282 #if TBB_PREVIEW_FLOW_GRAPH_FEATURES
283  case add_blt_pred:
284  my_predecessors.internal_add_built_predecessor(*(current->my_pred));
285  __TBB_store_with_release(current->status, SUCCEEDED);
286  break;
287  case del_blt_pred:
288  my_predecessors.internal_delete_built_predecessor(*(current->my_pred));
289  __TBB_store_with_release(current->status, SUCCEEDED);
290  break;
291  case blt_pred_cnt:
292  current->cnt_val = my_predecessors.predecessor_count();
293  __TBB_store_with_release(current->status, SUCCEEDED);
294  break;
295  case blt_pred_cpy:
296  my_predecessors.copy_predecessors(*(current->plist));
297  __TBB_store_with_release(current->status, SUCCEEDED);
298  break;
299 #endif /* TBB_PREVIEW_FLOW_GRAPH_FEATURES */
300  }
301  }
302  }

456  void handle_operations(queueing_port_operation* op_list) {
457  queueing_port_operation *current;
458  bool was_empty;
459  while(op_list) {
460  current = op_list;
461  op_list = op_list->next;
462  switch(current->type) {
463  case try__put_task: {
464  task *rtask = NULL;
465  was_empty = this->buffer_empty();
466  this->push_back(current->my_val);
467  if (was_empty) rtask = my_join->decrement_port_count(false);
468  else
469  rtask = SUCCESSFULLY_ENQUEUED;
470  current->bypass_t = rtask;
471  __TBB_store_with_release(current->status, SUCCEEDED);
472  }
473  break;
474  case get__item:
475  if(!this->buffer_empty()) {
476  this->copy_front(*(current->my_arg));
477  __TBB_store_with_release(current->status, SUCCEEDED);
478  }
479  else {
480  __TBB_store_with_release(current->status, FAILED);
481  }
482  break;
483  case res_port:
484  __TBB_ASSERT(this->my_item_valid(this->my_head), "No item to reset");
485  this->destroy_front();
486  if(this->my_item_valid(this->my_head)) {
487  (void)my_join->decrement_port_count(true);
488  }
489  __TBB_store_with_release(current->status, SUCCEEDED);
490  break;
491 #if TBB_PREVIEW_FLOW_GRAPH_FEATURES
492  case add_blt_pred:
493  my_built_predecessors.add_edge(*(current->pred));
494  __TBB_store_with_release(current->status, SUCCEEDED);
495  break;
496  case del_blt_pred:
497  my_built_predecessors.delete_edge(*(current->pred));
498  __TBB_store_with_release(current->status, SUCCEEDED);
499  break;
500  case blt_pred_cnt:
501  current->cnt_val = my_built_predecessors.edge_count();
502  __TBB_store_with_release(current->status, SUCCEEDED);
503  break;
504  case blt_pred_cpy:
505  my_built_predecessors.copy_edges(*(current->plist));
506  __TBB_store_with_release(current->status, SUCCEEDED);
507  break;
508 #endif /* TBB_PREVIEW_FLOW_GRAPH_FEATURES */
