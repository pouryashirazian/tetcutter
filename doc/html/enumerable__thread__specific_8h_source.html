175  template<ets_key_usage_type ETS_key_type>
176  void* ets_base<ETS_key_type>::table_lookup( bool& exists ) {
177  const key_type k = key_of_current_thread();
178 
179  __TBB_ASSERT(k!=0,NULL);
180  void* found;
181  size_t h = hash(k);
182  for( array* r=my_root; r; r=r->next ) {
183  call_itt_notify(acquired,r);
184  size_t mask=r->mask();
185  for(size_t i = r->start(h); ;i=(i+1)&mask) {
186  slot& s = r->at(i);
187  if( s.empty() ) break;
188  if( s.match(k) ) {
189  if( r==my_root ) {
190  // Success at top level
191  exists = true;
192  return s.ptr;
193  } else {
194  // Success at some other level. Need to insert at top level.
195  exists = true;
196  found = s.ptr;
197  goto insert;
198  }
199  }
200  }
201  }
202  // Key does not yet exist. The density of slots in the table does not exceed 0.5,
203  // for if this will occur a new table is allocated with double the current table
204  // size, which is swapped in as the new root table. So an empty slot is guaranteed.
205  exists = false;
206  found = create_local();
207  {
208  size_t c = ++my_count;
209  array* r = my_root;
210  call_itt_notify(acquired,r);
211  if( !r || c>r->size()/2 ) {
212  size_t s = r ? r->lg_size : 2;
213  while( c>size_t(1)<<(s-1) ) ++s;
214  array* a = allocate(s);
215  for(;;) {
216  a->next = r;
217  call_itt_notify(releasing,a);
218  array* new_r = my_root.compare_and_swap(a,r);
219  if( new_r==r ) break;
220  call_itt_notify(acquired, new_r);
221  if( new_r->lg_size>=s ) {
222  // Another thread inserted an equal or bigger array, so our array is superfluous.
223  free(a);
224  break;
225  }
226  r = new_r;
227  }
228  }
229  }
230  insert:
231  // Whether a slot has been found in an older table, or if it has been inserted at this level,
232  // it has already been accounted for in the total. Guaranteed to be room for it, and it is
233  // not present, so search for empty slot and use it.
234  array* ir = my_root;
235  call_itt_notify(acquired, ir);
236  size_t mask = ir->mask();
237  for(size_t i = ir->start(h);;i=(i+1)&mask) {
238  slot& s = ir->at(i);
239  if( s.empty() ) {
240  if( s.claim(k) ) {
241  s.ptr = found;
242  return found;
243  }
244  }
245  }

572  void advance_me() {
573  __TBB_ASSERT(my_segcont != NULL, NULL);
574  __TBB_ASSERT(outer_iter != my_segcont->end(), NULL); // not true if there are no inner containers
575  __TBB_ASSERT(inner_iter != outer_iter->end(), NULL); // not true if the inner containers are all empty.
