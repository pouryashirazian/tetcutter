91  int q;
92  T vmin[3], vmax[3], v;
93  for (q = X; q <= Z; q++) {
94  v = vert[q]; // -NJMP-
95  if (normal[q] > 0.0f)
96  {
97  vmin[q] = -maxbox[q] - v; // -NJMP-
98  vmax[q] = maxbox[q] - v; // -NJMP-

104  }
105  }

109  if (DOT(normal,vmax) >= 0.0f)

117 int IntersectBoxTriangle(T boxcenter[3], T boxhalfsize[3], T triverts[3][3]) {
118 
119  /* Use separating axis theorem to test overlap between triangle and box
120  need to test for overlap in these directions:
121  1) the {x,y,z}-directions (actually, since we use the AABB of the triangle
122  we do not even need to test these)
123  2) normal of the triangle
124  3) crossproduct(edge from tri, {x,y,z}-directin)
125  this gives 3x3=9 more tests
126  */
127  T v0[3], v1[3], v2[3];
128  T min, max, p0, p1, p2, rad, fex, fey, fez; // -NJMP- "d" local variable removed
129  T normal[3], e0[3], e1[3], e2[3];
130 
131  /* This is the fastest branch on Sun */
132  /* move everything so that the boxcenter is in (0,0,0) */
133 
134  SUB(v0, triverts[0], boxcenter);
135 
136  SUB(v1, triverts[1], boxcenter);
137 
138  SUB(v2, triverts[2], boxcenter);
139 
140  /* compute triangle edges */
141 
142  SUB(e0, v1, v0); /* tri edge 0 */
143 
144  SUB(e1, v2, v1); /* tri edge 1 */
145 
146  SUB(e2, v0, v2); /* tri edge 2 */
147 
148  /* Bullet 3: */
149 
150  /* test the 9 tests first (this was faster) */
151 
152  fex = fabsf(e0[X]);
153 
154  fey = fabsf(e0[Y]);
155 
156  fez = fabsf(e0[Z]);
157 
158  AXISTEST_X01(e0[Z], e0[Y], fez, fey);
159 
160  AXISTEST_Y02(e0[Z], e0[X], fez, fex);
161 
162  AXISTEST_Z12(e0[Y], e0[X], fey, fex);
163 
164  fex = fabsf(e1[X]);
165 
166  fey = fabsf(e1[Y]);
167 
168  fez = fabsf(e1[Z]);
169 
170  AXISTEST_X01(e1[Z], e1[Y], fez, fey);
171 
172  AXISTEST_Y02(e1[Z], e1[X], fez, fex);
173 
174  AXISTEST_Z0(e1[Y], e1[X], fey, fex);
175 
176  fex = fabsf(e2[X]);
177 
178  fey = fabsf(e2[Y]);
179 
180  fez = fabsf(e2[Z]);
181 
182  AXISTEST_X2(e2[Z], e2[Y], fez, fey);
183 
184  AXISTEST_Y1(e2[Z], e2[X], fez, fex);
185 
186  AXISTEST_Z12(e2[Y], e2[X], fey, fex);
187 
188  /* Bullet 1: */
189 
190  /* first test overlap in the {x,y,z}-directions */
191 
192  /* find min, max of the triangle each direction, and test for overlap in */
193 
194  /* that direction -- this is equivalent to testing a minimal AABB around */
195 
196  /* the triangle against the AABB */
197 
198  /* test in X-direction */
199 
200  FINDMINMAX(v0[X], v1[X], v2[X], min, max);
201 
202  if (min > boxhalfsize[X] || max < -boxhalfsize[X])
203  return 0;
204 
205  /* test in Y-direction */
206 
207  FINDMINMAX(v0[Y], v1[Y], v2[Y], min, max);
208 
209  if (min > boxhalfsize[Y] || max < -boxhalfsize[Y])
210  return 0;
211 
212  /* test in Z-direction */
213 
214  FINDMINMAX(v0[Z], v1[Z], v2[Z], min, max);
215 
216  if (min > boxhalfsize[Z] || max < -boxhalfsize[Z])
217  return 0;
218 
219  /* Bullet 2: */
220 
221  /* test if the box intersects the plane of the triangle */
222 
223  /* compute plane equation of triangle: normal*x+d=0 */
224 
225  CROSS(normal, e0, e1);
226 
227  // -NJMP- (line removed here)
228 
229  if (!PlaneBoxOverlap<T>(normal, v0, boxhalfsize))
230  return 0; // -NJMP-
231 
232  return 1; /* box and triangle overlaps */
233 }

254  outV = 0.0;
255  outT = 0.0;
256 
257  Vec3<T> p0 = Vec3<T>(&triverts[0]);
258  Vec3<T> p1 = Vec3<T>(&triverts[1]);
259  Vec3<T> p2 = Vec3<T>(&triverts[2]);
260  Vec3<T> o = Vec3<T>(&rayorigin[0]);
261  Vec3<T> d = Vec3<T>(&raydir[0]);
262 
263  Vec3<T> e1 = p1 - p0;
264  Vec3<T> e2 = p2 - p0;
265  Vec3<T> q = Vec3<T>::cross(d, e2);
266 
267  //Test Determinant
268  T a = Vec3<T>::dot(e1, q);
269  if(fabs(a) < EPSILON)
270  return 0;
271 
272  //Test U
273  T f = 1.0 / a;
274  Vec3<T> s = o - p0;
275  outU = f * Vec3<T>::dot(s, q);
276  if(outU < 0.0)
277  return 0;
278 
279  //Test V
280  Vec3<T> r = Vec3<T>::cross(s, e1);
281  outV = f * Vec3<T>::dot(d, r);
282  if((outV < 0.0) || (outU + outV > 1.0))
283  return 0;
284 
285  //Test T
286  outT = f * Vec3<T>::dot(e2, r);
287 
288  return 1;
