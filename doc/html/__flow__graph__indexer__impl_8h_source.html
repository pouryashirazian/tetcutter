197  void handle_operations(indexer_node_base_operation* op_list) {
198  indexer_node_base_operation *current;
199  while(op_list) {
200  current = op_list;
201  op_list = op_list->next;
202  switch(current->type) {
203 
204  case reg_succ:
205  my_successors.register_successor(*(current->my_succ));
206  __TBB_store_with_release(current->status, SUCCEEDED);
207  break;
208 
209  case rem_succ:
210  my_successors.remove_successor(*(current->my_succ));
211  __TBB_store_with_release(current->status, SUCCEEDED);
212  break;
213  case try__put_task: {
214  current->bypass_t = my_successors.try_put_task(*(current->my_arg));
215  __TBB_store_with_release(current->status, SUCCEEDED); // return of try_put_task actual return value
216  }
217  break;
218 #if TBB_PREVIEW_FLOW_GRAPH_FEATURES
219  case add_blt_succ:
220  my_successors.internal_add_built_successor(*(current->my_succ));
221  __TBB_store_with_release(current->status, SUCCEEDED);
222  break;
223  case del_blt_succ:
224  my_successors.internal_delete_built_successor(*(current->my_succ));
225  __TBB_store_with_release(current->status, SUCCEEDED);
226  break;
227  case blt_succ_cnt:
228  current->cnt_val = my_successors.successor_count();
229  __TBB_store_with_release(current->status, SUCCEEDED);
230  break;
231  case blt_succ_cpy:
232  my_successors.copy_successors(*(current->succv));
233  __TBB_store_with_release(current->status, SUCCEEDED);
234  break;
235 #endif /* TBB_PREVIEW_FLOW_GRAPH_FEATURES */
236  }
237  }
238  }
